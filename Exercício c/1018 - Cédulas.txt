#include <stdio.h>

int main() {
    int N;
    int cedulas[7] = {100, 50, 20, 10, 5, 2, 1};
    int quantidade[7];

    scanf("%d", &N);

    printf("%d\n", N);

    for (int i = 0; i < 7; i++) {
        quantidade[i] = N / cedulas[i];
        N = N % cedulas[i];
    }

    for (int i = 0; i < 7; i++) {
        printf("%d nota(s) de R$ %d,00\n", quantidade[i], cedulas[i]);
    }

    return 0;
}
